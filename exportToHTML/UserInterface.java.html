package GUI;
import net.sf.json.JSONObject;

import javax.swing.*;
import javax.swing.border.Border;
import java.awt.*;
import java.awt.event.ItemEvent;
import java.awt.event.ItemListener;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;
import java.awt.image.BufferedImage;

public class UserInterface extends JFrame {
    private JPanel jp = null;
    private JComboBox citys = null;
    private JButton search = null;
    private JTextArea result = null;
    public static String url="http://v.juhe.cn/weather/index";


    public UserInterface() {
        jp = new JPanel();
        String city_array[] = {"南京","北京"};
        citys = new JComboBox(city_array);
        result = new JTextArea("kk");
        result.setEditable(false);
        search = new JButton("SEARCH");


        jp.add(citys);
        jp.add(search);

        this.setBounds(100,100,100,100);
        this.add(result, BorderLayout.CENTER);
        this.add(jp,BorderLayout.SOUTH);
        this.setSize(700, 500);
        this.setTitle("WEATHER!");
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setVisible(true);

        search.addMouseListener(new MouseAdapter() {
            @Override
            public void mouseClicked(MouseEvent e) {
                JSONObject obj=JSONObject.fromObject(WeatherAPI.APIConnection.getConnection("南京"));
                JSONObject res=(JSONObject) obj.get("result");
                JSONObject sk=(JSONObject) res.get("sk");

                result.append("\n温度： "+sk.getString("temp"));
            }
        });

    }
    public static void main(String arg[]){
        UserInterface UI= new UserInterface();

    }
}
